import numpy as np
import matplotlib.pyplot as plt

def one_day_price_generator(n,s,a):
    sigma=0.0002 #s/np.sqrt(n)
    b=1-a
    P=[1]
    for i in range(1,n):
        w=np.random.normal(0,1)
        P.append(b+a*P[i-1]+sigma*w)
    return P

plt.plot(one_day_price_generator(1440,0.0002,1))
